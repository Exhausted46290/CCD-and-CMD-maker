#importing modules
import csv
import matplotlib.pyplot as plt
from matplotlib.ticker import MaxNLocator

with open('cep_objects.csv', 'r') as file:
    content = csv.reader(file)
    next(content, None)

    x_color = []
    y_color = []

    for row in content:
        try:
            w1 = float(row[10])   # W1 magnitude
            w2 = float(row[11])   # W2 magnitude
            j  = float(row[13])   # J magnitude
        except (ValueError, IndexError):
            continue

        x_color.append(w1 - w2)
        y_color.append(j - w2)

plt.figure(figsize=(10, 6), dpi=150)
plt.scatter(x_color, y_color, s=10, alpha=0.6, edgecolors='none')

plt.xlabel('W1 - W2 (mag)')
plt.ylabel('J - W2 (mag)')
plt.title('Color-Color Diagram for Cepheid Candidates')

ax = plt.gca()
ax.xaxis.set_major_locator(MaxNLocator(5))
ax.yaxis.set_major_locator(MaxNLocator(5))

plt.tight_layout()
plt.show()
