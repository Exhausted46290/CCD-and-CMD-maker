import matplotlib.pyplot as plt
import csv
from matplotlib.ticker import MaxNLocator

with open('cep_objects.csv', 'r') as file:
    content = csv.reader(file)
    next(content, None)  # skip header

    x_color = []  # W1 - W2 color
    y_mag = []    # W1 magnitude

    for row in content:
        try:
            w1 = float(row[10])   # W1 magnitude (column 10)
            w2 = float(row[11])   # W2 magnitude (column 11)
            j = float(row[14])
        except (ValueError, IndexError):
            continue

        color = j - w2   # W1 - W2
        mag = w2          # Use W1 as the magnitude axis

        x_color.append(color)
        y_mag.append(mag)

plt.figure(figsize=(10, 6), dpi=150)
plt.scatter(x_color, y_mag, s=10, alpha=0.6, edgecolors='none')

plt.xlabel('J-W2 (mag)')
plt.ylabel('W2 (mag)')
plt.title('Color-Magnitude Diagram for Cepheid Candidates (J-W2 vs W2)')

ax = plt.gca()
ax.xaxis.set_major_locator(MaxNLocator(5))
ax.yaxis.set_major_locator(MaxNLocator(5))
ax.invert_yaxis()

plt.tight_layout()
plt.show()
